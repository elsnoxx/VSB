def spocti_krychli(soubor_souradnice):
    with open(soubor_souradnice, 'r') as file:
        rows = file.readlines()
    
    x, y, z = [], [], []

    for row in rows:
        stripped = row.strip().split(',')
        x1, y1, z1 = map(int, stripped)

        x.append(x1)
        y.append(y1)
        z.append(z1)

    #print("Obsah proměnné x1:", x)
    #print("Obsah proměnné y1:", y)
    #print("Obsah proměnné z1:", z)

    x_min, x_max = min(x), max(x)
    y_min, y_max = min(y), max(y)
    z_min, z_max = min(z), max(z)

    x_size = abs(x_min) + abs(x_max)
    y_size = abs(y_min) + abs(y_max)
    z_size = abs(z_min) + abs(z_max)

    return x_size * y_size * z_size
